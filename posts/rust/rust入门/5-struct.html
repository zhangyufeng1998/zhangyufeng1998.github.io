                                                                        # 5.1 定义和实例化struct

                                                                        # 什么是struct

                                                                        struct，结构体

                                                                        • 自定义的数据类型
                                                                        • 为相关联的值命名，打包成有意义的组合

                                                                        # 定义struct

                                                                        • 使用struct关键字，并为整个struct命名

                                                                        • 在花括号内，为所有字段Field定义名称和类型

                                                                        • 例如:

                                                                          struct User{
    userbane: String,
    email: String,
    sign_in_count: u64,
    active: bool,
}

                                                                        # 实例化struct

                                                                        • 想要使用stuct，需要创建stuct实例：

                                                                          • 为每个字段指定具体值
                                                                          • 无需按声明的顺序进行指定

                                                                        • 例如

                                                                          struct User{
    username: String,
    email: String,
    sign_in_count: u64,
    active: bool,
}

fn main(){
    print!("Hello,world!");
    let user1 = User{
        email: String::from("907392790@qq.com"),
        username: String::from("zyf"),
        active: true,
        sign_in_count: 1006,
    };
}

                                                                        # 取得struct里面的某个值

                                                                        • 使用点标记法

                                                                        • print!("{}",user1.username);

                                                                        • 修改某值user1.email = String::from("another@example.com");需要在实例化struct时声明mutlet mut user1 = User{...};

                                                                          • 注一旦struct的实例是可变的，那么实例中所有字段都是可变的

                                                                        # struct作为函数的返回值

                                                                        fn build_user(email: String,username: String) -> User{
    User{
        email: email,
        username: username,
        active: true,
        sign_in_count: 1,
    }
}

                                                                        提示

                                                                        上面代码中Uer{}后面没有分号

                                                                        在rust中函数的最后一个表达式就是它的返回值。

                                                                        # 字段初始化简写

                                                                        当字段名与字段值对应变量名相同时，就可以使用字段初始化简写的方式

                                                                        # struct 更新语法

                                                                        当你想基于某个struct实例来创建一个新的实例的时候，可以使用struct更新语法

                                                                        正常写法：

                                                                        let user2 = User{
    email: String::from("907392790@11.com"),
    username: String::from("zyf"),
    acitve: user1.active,
    sign_in_count: user1.sign_in_count,
};



                                                                         
                                                                         

                                                                        更新语法：

                                                                        let user2 = User{
    email: String::from("907392790@11.com"),
    username: String::from("zyf"),
    ..user1
};



                                                                         

                                                                        # Tuple struct

                                                                        • 可以定义类似tuple的struct，叫做tuple struct
                                                                          • Tuple struct整体有个名，但里面的元素没有名
                                                                          • 适用：想给整个tuple起名，并让他不同于其他tuple，而且不需要给每个元素起名
                                                                        • 定义tuple struct：使用struct关键字，后面是名字，以及里面的元素的类型
                                                                        • 例子
                                                                          struct Color(i32,i32,i32);
struct Point(i32,i32,i32);
let black = Color(0,0,0);
let origin = Point(0,0,0);
                                                                        • 注black and origin is different type,是不同tuple struct的实例

                                                                        # struct 数据的所有权

                                                                        struct User{
    username: String,
    email: String,
    sign_in_count: u64,
    active: bool,
}
                                                                        • 这里的字段使用了String而不是&str：

                                                                        • struct里也可以存放引用，但是这需要使用生命周期（后续讲）

                                                                        # 5.2 struct 例子

                                                                        使用rust写一个计算长方形面积的例子

                                                                        简单写法：

                                                                        fn main() {
    let w = 30;
    let l = 50;
    print!("{}",area(w,l));
}
fn area(width: u32, length: u32) -> u32 {
    width * length
}

                                                                        使用元组写：

                                                                        fn main() {
    // 声明以个长方形元组
    let rect = (30, 50);
    print!("{}", area(rect));
}
fn area(dim: (u32, u32)) -> u32 {
    dim.0 * dim.1
}

                                                                        使用struct重构:

                                                                        struct Rectangle {
    width: u32,
    length: u32,
}
fn main() {
    let rect = Rectangle {
        width: 30,
        length: 50,
    };
    print!("{}", area(&rect));
}
fn area(rect: &Rectangle) -> u32 {
    rect.width * rect.length
}

                                                                        # 5.3 struct方法

                                                                        # struct的方法

                                                                        • 方法和函数类似：fn关键字、名称、参数、返回值
                                                                        • 方法与函数的不同之处
                                                                          • 方法是在struct（或enum、trait对象）的上下文定义的
                                                                          • 方法的第一个参数是self，表示方法被调用的struct实例

                                                                        # 定义方法

                                                                        • 在impl块里定义方法
                                                                        • 方法的第一个参数可以是&self，也可以获得其所有权或可变借用。和其他参数一样
                                                                        • 更良好的代码结构
                                                                        • 例子：
                                                                        #[derive(Debug)]
struct Rectangle {
    width: u32,
    length: u32,
}
impl Rectangle{
    fn area(&self) -> u32{
        self.width * self.length
    }
}
fn main() {
    let rect = Rectangle {
        width: 30,
        length: 50,
    };
    println!("{}",rect.area());
    print!("{:#?}",rect);
}

                                                                        # 方法调用的运算符

                                                                        • 在c/c++中：object->someting()和(*object).something()一样
                                                                        • rust没有->运算符
                                                                        • rust会自动引用或解引用
                                                                          • 在调用方法的时候会发生这种行为
                                                                        • 在调用方法时，rust会根据情况自动添加&、&mut或*(解引用)以便object可以匹配方法的签名
                                                                        • 下面两行代码效果相同
                                                                          • p1.distance(&p2);
                                                                          • (&p1).distance(&p2);

                                                                        # 方法参数

                                                                        例子

                                                                        struct Rectangle {
    width: u32,
    length: u32,
}
impl Rectangle{
    fn area(&self) -> u32{
        self.width * self.length
    }
    // 定义一个方法，判断一个矩形是否能容纳另一个矩形
    fn can_hold(&self,other: &Rectangle) -> bool{
        self.width > other.width && self.length > other.length
    }
}
fn main() {
    let rect1 = Rectangle {
        width: 30,
        length: 50,
    };
    let rect2 = Rectangle {
        width: 10,
        length: 40,
    };
    let rect3 = Rectangle {
        width: 35,
        length: 55,
    };
    println!("矩形1的面积为{}",rect1.area());
    println!("矩形2的面积为{}",rect2.area());
    println!("矩形3的面积为{}",rect3.area());
    println!("{}",rect1.can_hold(&rect2));
    println!("{}",rect1.can_hold(&rect3));
}

矩形2的面积为400
矩形3的面积为1925
true
