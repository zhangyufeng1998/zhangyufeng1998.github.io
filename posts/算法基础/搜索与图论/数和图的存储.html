                # 树和图的存储

                树是一种特殊的图（无环连通图）

                • 有向图
                  • a -> b
                • 无向图
                  • a - b
                  • 特殊的有向图

                两种存储方式：

                • 邻接矩阵（用的比较少，浪费空间，空间复杂度为n^2，多用于存储稠密图）
                  • 不能存储重边
                  • 二维数组g[a,b] a -> b
                    • true代表有连通，false代表无连通
                • 邻接表（用的比较多，单链表）
                  • 和拉链法存储hash一样
                  • 开启n个单链表，每个点上都有一个单链表，存储这个点可以走到哪个点

                #include <cstring>
#include <iostream>
#include <algorithm>

using namespace std;
const int N = 100010, M = N * 2;
// h: head, e: edge, ne: next edge, idx: index
int h[N], e[M], ne[M], idx;
bool st[N]; // st: state which means whether the node is visited or not
int n, m;
int ans = N;

void add(int a, int b) // 插入一条a指向b的边
{
    // a所对应的邻接表里插入一个节点b
    e[idx] = b;
    ne[idx] = h[a];
    h[a] = idx++;
}

// 以u为根的子树中点的数量
int dfs(int u) {
    st[u] = true;  // 标记u节点已经被遍历过

    int sum = 1, res = 0;
    // Access each child node of u
    for (int i = h[u]; i != -1; i = ne[i]) {
        int j = e[i];
        if (!st[j]) {
            int s = dfs(j);
            res = max(res, s);
            sum += s;
        }
    }

    res = max(res, n - sum);
    ans = min(ans, res);
    return sum;
}

int main() {
    cin >> n;
    memset(h, -1, sizeof h); // n个头节点 全部初始化为 -1
    for (int i = 0; i < n - 1; i++) {
        int a, b;
        cin >> a >> b;
        add(a, b);
        add(b, a);
    }
    dfs(1);

    cout << ans << endl;

    return 0;
}

                package com.zyf;

import java.util.Arrays;
import java.util.Scanner;

public class Tree {
    static int N = 100010, M = N * 2, idx, n;
    static int[] h = new int[N], e = new int[M], ne = new int[M];
    static boolean[] st = new boolean[N];
    static int ans = N;

    public static void add(int a, int b) {
        e[idx] = b;
        ne[idx] = h[a];
        h[a] = idx++;
    }

    public static int dfs(int u) {
        int res = 0;
        st[u] = true;
        int sum = 1;

        // 单链表遍历
        for (int i = h[u]; i != -1; i = ne[i]){
            int j = e[i];
            if (!st[j]) {
                int s = dfs(j);
                res = Math.max(res, s);
                sum += s;
            }
        }

        res = Math.max(res,n-sum);
        ans = Math.min(ans,res);
        return sum;
    }

    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        n = sc.nextInt();
        Arrays.fill(h, -1);  // 初始化邻接表
        for (int i = 0; i < n - 1; i++) {
            int a = sc.nextInt(), b = sc.nextInt();
            add(a, b);
            add(b, a);
        }

        dfs(1);
        System.out.println(ans);
    }
}


                const N: usize = 100010;
const M: usize = N * 2;

struct Edge {
    e: usize,
    ne: usize,
}

fn add(h: &mut Vec<usize>, e: &mut Vec<Edge>, a: usize, b: usize) {
    e.push(
        Edge { e: b, ne: h[a] }
    );
    h[a] = e.len() - 1;
}

fn dfs(h: &[usize], e: &[Edge], st: &mut Vec<bool>, n: usize, u: usize, ans: &mut usize) -> usize {
    st[u] = true;
    let mut sum = 1;
    let mut res = 0;
    for i in (0..n).filter(|&i| h[u + i] != usize::MAX) {
        let j = e[h[u + i]].e;
        if !st[j] {
            let s = dfs(h, e, st, n, j, ans);
            res = res.max(s);
            sum += s;
        }
    }
    res = res.max(n - sum);
    *ans = (*ans).min(res);
    sum
}

fn main() {
    let mut h = vec![usize::MAX; N];
    let mut e = Vec::with_capacity(M);
    let mut st = vec![false; N];
    let mut ans = N;

    let mut input = String::new();
    std::io::stdin().read_line(&mut input).unwrap();
    let n: usize = input.trim().parse().unwrap();

    for _ in 0..n - 1 {
        input.clear();
        std::io::stdin().read_line(&mut input).unwrap();
        let nums: Vec<usize> = input
            .split_whitespace()
            .map(|x| x.parse().unwrap())
            .collect();
        let a = nums[0];
        let b = nums[1];
        add(&mut h, &mut e, a, b);
        add(&mut h, &mut e, b, a);
    }

    dfs(&h, &e, &mut st, n, 1, &mut ans);

    println!("{}", ans);
}
