              # 走迷宫

              #include <iostream>
#include <algorithm>
#include <cstring>

using namespace std;

typedef pair<int, int> PII;

const int N = 110;

int n, m;
int g[N][N]; // 存储地图
int d[N][N]; // 每一个点到起点的距离
PII q[N * N]; // 队列

int bfs() {
    int hh = 0, tt = 0;  // 队列头和队列尾
    q[0] = {0, 0};  // 起点
    memset(d, -1, sizeof d);  // 初始化所有点到起点的距离为-1
    d[0][0] = 0;  // 起点到起点的距离为0
    int dy[4] = {0, 1, 0, -1};
    int dx[4] = {-1, 0, 1, 0};  // 方向数组
    while (hh <= tt) {
        auto t = q[hh++];
        for (int i = 0; i < 4; i++) {
            int x = t.first + dx[i], y = t.second + dy[i];
            if (x >= 0 && x < n && y >= 0 && y < m && g[x][y] == 0 && d[x][y] == -1) {
                d[x][y] = d[t.first][t.second] + 1;
                q[++tt] = {x, y};
            }
        }
    }
    return d[n - 1][m - 1];
}

int main() {
    cin >> n >> m;
    for (int i = 0; i < n; i++)
        for (int j = 0; j < m; j++)
            cin >> g[i][j];
    cout << bfs() << endl;
    return 0;
}

              package com.zyf;

import java.io.*;
import java.util.*;

public class Main {
    static int N = 110, hh, tt, n, m;
    static int[][] g = new int[N][N]; // Used to store maps
    static int[][] d = new int[N][N]; // Used to store the distance from the starting point to the current point
    static PII[] q = new PII[N * N]; // Used to store the coordinates of the current point

    public static int bfs() {
        hh = 0;
        tt = -1;  // The queue's head node = 0 and tail node = -1
        d[0][0] = 0; // The distance from the starting point to the starting point is 0
        q[++tt] = new PII(0, 0); // Add the starting point to the queue
        int[] dx = {0, 1, 0, -1}; // Four directions
        int[] dy = {1, 0, -1, 0}; // Four directions
        while (hh <= tt) {
            PII t = q[hh++]; // Take out the head node of the queue
            for (int i = 0; i < 4; i++){
                int x = t.first + dx[i], y = t.second + dy[i]; // The next point coordinates
                if (x >= 0 && x < n && y >= 0 && y < m && g[x][y] == 0 && d[x][y] == -1) { // If the next point is in the map and is not blocked and has not been visited
                    d[x][y] = d[t.first][t.second] + 1; // The distance from the starting point to the next point is the distance from the starting point to the current point +1
                    q[++tt] = new PII(x, y); // Add the next point to the queue
                }
            }
        }
        return d[n - 1][m - 1]; // Return the distance from the starting point to the end point
    }

    public static void main(String[] args)throws IOException {
        BufferedReader re = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter wt = new BufferedWriter(new OutputStreamWriter(System.out));
        String[] s = re.readLine().split(" ");
        n = Integer.parseInt(s[0]);
        m = Integer.parseInt(s[1]);
        for(int i = 0 ; i < n ;  i ++ ){
            String[] st = re.readLine().split(" ");
            for(int j = 0 ; j < m ;j ++ ){
                g[i][j] = Integer.parseInt(st[j]);
                d[i][j] = -1;
            }
        }
        System.out.println(bfs());
        wt.close();
    }
}
class PII {
    int first, second;

    public PII(int first, int second) {
        this.first = first;
        this.second = second;
    }
}
